    interface Calculator {
        float add(in float a, in float b);
        float subtract(in float a, in float b);
        float multiply(in float a, in float b);
        float divide(in float a, in float b) raises(DivisionByZero);

//CalculatorClient.java

import CalculatorApp.*;
import org.omg.CORBA.*;
import org.omg.CosNaming.*;

public class CalculatorClient {
    public static void main(String args[]) {
        try {
            // Initialize ORB
            ORB orb = ORB.init(args, null);

            // Get reference to Naming Service
            org.omg.CORBA.Object objRef = orb.resolve_initial_references("NameService");
            NamingContextExt ncRef = NamingContextExtHelper.narrow(objRef);

            // Resolve Calculator object
            Calculator calculator = CalculatorHelper.narrow(ncRef.resolve_str("Calculator"));

            // Perform operations
            System.out.println("Addition (10 + 5): " + calculator.add(10, 5));
            System.out.println("Subtraction (10 - 5): " + calculator.subtract(10, 5));
            System.out.println("Multiplication (10 * 5): " + calculator.multiply(10, 5));

            try {
                System.out.println("Division (10 / 2): " + calculator.divide(10, 2));
                System.out.println("Division (10 / 0): " + calculator.divide(10, 0)); // Should raise exception
            } catch (DivisionByZero e) {
                System.out.println("Error: Division by zero is not allowed!");
            }

        } catch (Exception e) {
            System.err.println("Client exception: " + e.getMessage());
            e.printStackTrace();
        }
    }
}



//CalculatorServer.java

import CalculatorApp.*;
import org.omg.CORBA.*;
import org.omg.CosNaming.*;
import org.omg.PortableServer.*;

class CalculatorImpl extends CalculatorPOA {
    private ORB orb;

    public void setORB(ORB orb_val) {
        orb = orb_val;
    }

    public float add(float a, float b) {
        return a + b;
    }

    public float subtract(float a, float b) {
        return a - b;
    }

    public float multiply(float a, float b) {
        return a * b;
    }

    public float divide(float a, float b) throws DivisionByZero {
        if (b == 0) throw new DivisionByZero();
        return a / b;
    }
}

public class CalculatorServer {
    public static void main(String args[]) {
        try {
            // Initialize ORB
            ORB orb = ORB.init(args, null);
            
            // Get reference to RootPOA and activate POA Manager
            POA rootpoa = POAHelper.narrow(orb.resolve_initial_references("RootPOA"));
            rootpoa.the_POAManager().activate();

            // Create servant and register it with ORB
            CalculatorImpl calculator = new CalculatorImpl();
            calculator.setORB(orb);
            org.omg.CORBA.Object ref = rootpoa.servant_to_reference(calculator);
            Calculator href = CalculatorHelper.narrow(ref);

            // Bind object to Naming Service
            org.omg.CORBA.Object objRef = orb.resolve_initial_references("NameService");
            NamingContextExt ncRef = NamingContextExtHelper.narrow(objRef);
            NameComponent path[] = ncRef.to_name("Calculator");
            ncRef.rebind(path, href);

            System.out.println("Calculator Server is running...");
            orb.run();
        } catch (Exception e) {
            System.err.println("Server exception: " + e.getMessage());
            e.printStackTrace();
        }
    }
}

## 🎓 ORAL EXAM QUESTIONS WITH ANSWERS

### Basic Level Questions
1. **What is CORBA and why is it important in distributed systems?**
   
   **Answer**:
   CORBA (Common Object Request Broker Architecture) is a standard for distributed object computing:
   - Enables objects written in different languages to communicate
   - Platform and OS independent
   - Standardized by OMG (Object Management Group)
   - Allows C++, Java, Python, C# objects to interoperate seamlessly
   
   Importance:
   - Solves heterogeneous system integration problem
   - Enterprise-grade reliability (proven in telecom, finance)
   - Single unified interface regardless of language/platform
   - Legacy system integration capability

2. **What is the main advantage of CORBA over RMI?**
   
   **Answer**:
   **Main Advantage: Language Independence**
   ```
   RMI: Java ←→ Java only
   CORBA: Java ←→ C++ ←→ Python ←→ C# (any language with CORBA binding)
   ```
   
   Specific advantages:
   - Can write server in C++, client in Java, both work seamlessly
   - Ideal for heterogeneous enterprises
   - Single standard interface (IDL) across languages
   - Large existing CORBA systems to integrate with
   
   CORBA also:
   - Platform independent (Windows/Linux/Unix/mainframe)
   - Hierarchical naming service (vs flat RMI registry)
   - More mature (longer track record)

3. **What is IDL (Interface Definition Language)?**
   
   **Answer**:
   IDL is a language-neutral specification for interfaces:
   - Separate from any programming language
   - Defines method signatures, parameters, return types, exceptions
   - Compiler generates language-specific code (stubs, skeletons)
   - Single IDL can generate Java, C++, Python implementations
   
   Example:
   ```idl
   interface Calculator {
       float add(in float a, in float b);
       float divide(in float a, in float b) raises(DivisionByZero);
   };
   ```
   
   Why separate?
   - Ensures interoperability (common contract)
   - Compiler verifies consistency across languages
   - Forces explicit interface design
   - Enables code generation

4. **What are the key components of CORBA architecture?**
   
   **Answer**:
   1. **ORB** (Object Request Broker): Middleware that handles communication
   2. **Stubs**: Client-side proxies (generated from IDL)
   3. **Skeletons**: Server-side proxies (generated from IDL)
   4. **POA** (Portable Object Adapter): Manages server-side objects
   5. **Naming Service**: Directory service for finding objects
   6. **Interface Repository**: Stores interface definitions
   7. **GIOP/IIOP**: Protocol for inter-ORB communication
   8. **IDL Compiler**: Generates language-specific files
   
   These work together to enable transparent distributed communication.

5. **What is an ORB (Object Request Broker)?**
   
   **Answer**:
   ORB is the communication middleware in CORBA:
   - Creates and manages object references
   - Routes method requests from client to server
   - Handles marshalling/unmarshalling (using CDR format)
   - Uses IIOP protocol for network communication
   - Manages object lifecycle
   
   Process:
   ```
   Client stub → ORB → Network (IIOP) → Server ORB → Skeleton → Servant
   ```
   
   ORB initialization:
   ```java
   ORB orb = ORB.init(args, null);
   ```
   After this, ORB handles all communication details.

### Intermediate Level Questions
6. **Explain the difference between CORBA and RMI.**
   
   **Answer**:
   | Feature | CORBA | RMI |
   |---------|-------|-----|
   | **Language Support** | Multi-language (C++, Java, Python, C#) | Java only |
   | **Platform** | Any OS (Windows, Linux, Unix, mainframe) | Any JVM (OS-independent via Java) |
   | **Interface** | IDL (separate, compiled) | Java interface (built-in) |
   | **Learning Curve** | Steep (IDL, compilation, ORB) | Easy (looks like Java) |
   | **Scalability** | 100s-1000s of objects | 10s-100s objects |
   | **Performance** | Slower (complex protocol) | Faster (simplified) |
   | **Ideal For** | Heterogeneous enterprises | Java systems |
   | **Industry** | Legacy, finance, telecom | Web services (declining) |

7. **What is the Naming Service in CORBA? How is it different from RMI Registry?**
   
   **Answer**:
   **Naming Service**:
   - Hierarchical directory (like file system)
   - Can have nested contexts (Services/Banking/Accounts)
   - More flexible than flat structure
   - Standardized by CORBA spec
   
   **RMI Registry**:
   - Flat namespace (just name → object)
   - Cannot have hierarchies
   - Simpler but less scalable
   - Java-specific
   
   Example:

   ```
   RMI:
   ```

   CORBA better for large systems with hierarchical organization.

8. **What is POA (Portable Object Adapter)?**
   
   **Answer**:
   POA manages server-side objects:
   - Converts servants (implementation objects) to CORBA references
   - Routes incoming requests to correct servant
   - Manages object lifecycle (creation, activation, deactivation)
   - Provides portability across ORB implementations
   
   Key operations:
   ```java
   POA rootpoa = POAHelper.narrow(orb.resolve_initial_references("RootPOA"));
   rootpoa.the_POAManager().activate();  // Activate POA
   Object ref = rootpoa.servant_to_reference(servant);  // Convert servant
   ```
   
   Why important:
   - Enables same code on different ORBs
   - Automatic stub/skeleton handling
   - Request dispatching mechanism

### Advanced Level Questions
11. **Compare CORBA marshalling with RMI marshalling.**
    
    **Answer**:
    **CORBA Marshalling**:
    - Uses CDR (Common Data Representation)
    - Binary format (compact, efficient)
    - Handles byte alignment automatically
    - Supports big-endian and little-endian
    - Handled by GIOP protocol layer
    - Works across all languages
    
    **RMI Marshalling**:
    - Uses Java serialization
    - More verbose than CDR
    - Java-specific object encoding
    - Less efficient but more feature-rich
    - Supports Java object graphs
    - Only works with Java
    
    Comparison:
    ```
    CORBA CDR: Compact, standardized, language-agnostic
    RMI Serialization: Complete, Java-rich, Java-only
    
    Example: Integer 300
    CDR: 2C 01 (hex) - 2 bytes
    RMI: ac f3 17 f6 06 01 01 2C 01 (more overhead)
    ```
    
    CDR is more efficient for network transmission.

12. **How does CORBA handle method invocation across different languages?**
    
    **Answer**:
    Process:
    ```
    C++ Client calls Java Server:
    
    1. C++ stub calls local method (looks local to C++)
    2. C++ stub marshals parameters using CDR
    3. IIOP sends over network
    4. Java skeleton receives IIOP packet
    5. Java skeleton unmarshals CDR to Java objects
    6. Java method called with native Java types
    7. Method executes, returns Java result
    8. Java skeleton marshals result to CDR
    9. IIOP sends result back
    10. C++ stub unmarshals to C++ types
    11. C++ code receives result
    ```
    
    IDL as Bridge:
    ```idl
    interface Calculator {
        float add(in float a, in float b);
    }
    ```
    
    Both C++ and Java implement this same interface.
    CDR format ensures all understand same data.

13. **What is the difference between oneway and regular method calls in CORBA?**
    
    **Answer**:
    **Regular (Synchronous)**:
    ```idl
    float add(in float a, in float b);
    ```
    - Client waits for response
    - Receives return value
    - Exception handling possible
    - Blocking call
    - Used for most operations
    
    **Oneway (Asynchronous)**:
    ```idl
    oneway void logEvent(in string message);
    ```
    - Client doesn't wait for response
    - No return value (always void)
    - No exception feedback
    - Non-blocking call
    - Used for notifications, logging
    
    When to use:
    - Oneway: Notifications, fire-and-forget
    - Regular: When you need result/confirmation

14. **Explain the CORBA object model and how it compares to OOP.**
    
    **Answer**:
    **CORBA Object Model**:
    - Objects are distributed (not in same process)
    - Accessed through references (not pointers)
    - Method invocation is remote (has latency)
    - Objects can fail independently
    - No direct memory access
    - Pass-by-value semantics for primitives
    
    **Comparison to Local OOP**:
    ```
    Local OOP:
    obj.method() → immediate execution → result
    
    CORBA:
    obj.method() → network → remote execution → network → result
    Time: microseconds → milliseconds
    ```
    
    Differences:
    - **Latency**: CORBA has ~1000x overhead
    - **Reliability**: Network can fail
    - **Concurrency**: Multiple clients, server-side synchronization
    - **State**: Only server holds state, clients are stateless
    - **Garbage Collection**: Server manages, not local GC

    2. **Object injection**: Malicious objects via deserialization
    3. **Unauthorized access**: No built-in authentication
    4. **Man-in-the-middle**: Attacker intercepts traffic
    
    **Mitigation Strategies**:
    1. **IIOPS** (IIOP Secure):
       ```
       Use SSL/TLS for encryption
       java -Dcom.sun.CORBA.security=true ...
       ```
    
    2. **Authentication**:
       ```idl
       interface SecureService {
           boolean login(in string user, in string password);
           Result getSecureData(in string sessionToken);
       };
       ```
    
    3. **Authorization**:
       - Check permissions on server
       - Role-based access control
       - Fine-grained method permissions
    
    4. **Firewalls**:
       - Limit IIOP port access
       - Whitelist trusted hosts
    
    5. **Serialization Filters**:
       - Restrict class deserialization
       - Validate object types before using

17. **How would you implement a CORBA-based chat system where multiple clients communicate?**
    
    **Answer**:
    Using callbacks/observers:
    ```idl
    module ChatSystem {
        interface ChatListener {
            oneway void onMessage(in string user, in string message);
        };
        
        interface ChatServer {
            void registerListener(in ChatListener listener);
            void sendMessage(in string user, in string message);
        };
    };
    ```
    
    Implementation:
    ```java
    class ChatServerImpl extends ChatServerPOA {
        List<ChatListener> listeners = new ArrayList<>();
        
        public void registerListener(ChatListener listener) {
            listeners.add(listener);
        }
        
        public void sendMessage(String user, String message) {
            for (ChatListener l : listeners) {
                l.onMessage(user, message);  // Callback to clients
            }
        }
    }
    ```
    
    How it works:
    1. Client registers as ChatListener with server
    2. Client1 sends message
    3. Server calls onMessage() on all registered listeners
    4. All clients receive notification via callback

    **Answer**:
    Scenario:
    ```

    ├─ Existing clients with object references: CONTINUE WORKING
    ├─ New clients trying Naming.lookup(): FAIL
    └─ New services can't register
    ```
    
    Solutions:
    1. **High Availability**:
       ```
       Multiple Naming Service instances
       Primary ← → Replica 1 ← → Replica 2
       If primary fails, promote replica
       ```
    
    2. **Embedded References**:
       ```
       Clients cache object references
       Use cached reference if Naming Service down
       Fallback to Naming Service on retry
       ```
    
    3. **Multiple Naming Contexts**:
       ```
       Each region has local Naming Service
       Hierarchy of naming services
       ```
    
    4. **Persistence**:
       ```
       Backup Naming Service data regularly
       Restore from backup on failure
       ```

## 🔧 DISTRIBUTED SYSTEMS CONCEPTS & THEORY

### 1. **CORBA (Common Object Request Broker Architecture)**

#### **Definition**
CORBA is a standardized framework for building distributed applications that integrate objects written in different programming languages and running on different platforms. It provides language and platform independence through a standardized interface definition language (IDL).

#### **Why CORBA?**
- **Language-Independent**: C++, Java, Python, C#, etc. can interoperate
- **Platform-Independent**: Windows, Linux, Unix, mainframes work together
- **Industry Standard**: OMG (Object Management Group) specification
- **Object-Oriented**: Distributed objects, not just procedures
- **Vendor Neutral**: Multiple ORB implementations available
- **Enterprise-Ready**: Mature, widely adopted in large systems

#### **When to Use CORBA?**
✅ Heterogeneous enterprise systems
✅ Multi-language integration (C++, Java, Python together)
✅ Legacy system integration
✅ Complex distributed systems
✅ High-reliability systems (finance, telecom)

❌ Simple web services (use REST)
❌ Microservices (use gRPC/REST)
❌ Real-time systems with strict latency requirements
❌ Mobile applications
❌ Cloud-native applications

#### **CORBA vs RMI - Comparison**

| Aspect | CORBA | RMI |
|--------|-------|-----|
| **Language Support** | Multi-language | Java only |
| **Platform Support** | Multi-platform | All (JVM-based) |
| **Interface Definition** | IDL (separate) | Java interface |
| **Learning Curve** | Steep | Easy |
| **Complexity** | High | Low |
| **Performance** | Slower | Faster |
| **Industry Adoption** | Enterprise | Declining |
| **Best For** | Large heterogeneous systems | Java-to-Java systems |

#### **CORBA Architecture Overview**
```
┌─────────────────────────────────────────────────────────────┐
│                    CORBA SYSTEM OVERVIEW                    │
├─────────────────────────────────────────────────────────────┤
│                                                              │
│  ┌─────────────────┐          ┌──────────────────┐         │
│  │   CLIENT SIDE   │          │  SERVER SIDE     │         │
│  │   (C++/Java)    │          │  (Java/C++)      │         │
│  └────────┬────────┘          └─────────┬────────┘         │
│           │                             │                   │
│        ┌──▼──────────┐           ┌──────▼─────┐            │
│        │ IDL Stubs   │           │  Skeletons │            │
│        └──┬──────────┘           └──────┬─────┘            │
│           │                             │                   │
│        ┌──▼────────────────────────────▼──┐               │
│        │         ORB (Object Request       │               │
│        │         Broker) - Network         │               │
│        │  Protocol: IIOP (TCP/IP)          │               │
│        │  Data Encoding: CDR               │               │
│        └──┬────────────────────────────┬──┘               │
│           │                            │                   │
│        ┌──▼─────────┐          ┌──────▼──────┐           │
│        │   NamingService  ◄────┤   POA &     │           │
│        │   (Service Discovery) │  Servants   │           │
│        └────────────┘          └─────────────┘           │
│                                                              │
└─────────────────────────────────────────────────────────────┘
```

#### **How CORBA Works - Step by Step**
```
1. IDL Definition
   ├─ Interface specification (language-independent)
   └─ Compiler generates stubs, skeletons, helpers

2. Implementation
   ├─ Implement interface in target language
   └─ Create Servant class

3. Server Startup
   ├─ Initialize ORB
   ├─ Create and register POA
   ├─ Instantiate servant
   └─ Register with Naming Service

4. Client Lookup
   ├─ Initialize ORB
   ├─ Query Naming Service
   └─ Get remote object reference

5. Method Invocation
   ├─ Call method on stub (looks local)
   ├─ Stub marshals parameters using CDR
   ├─ ORB sends IIOP packet to server
   ├─ Server skeleton unmarshals data
   ├─ Servant executes method
   ├─ Result marshaled and sent back
   └─ Client receives and unmarshals

6. Result Delivery
   ├─ Stub unmarshals response
   └─ Application receives result
```

### 2. **IDL (Interface Definition Language)**

#### **Definition**
IDL is a language-neutral way to specify interfaces that CORBA objects must implement. It is independent of any programming language.

#### **Purpose**
- **Contract Definition**: Specifies what methods are available
- **Language Mapping**: Defines how IDL maps to different languages
- **Compilation**: Compiler generates language-specific stubs and skeletons
- **Interoperability**: Ensures consistency across heterogeneous systems

#### **IDL Structure**
```idl
module ModuleName {
    // Type definitions
    typedef long myInteger;
    
    // Exceptions
    exception DivisionByZero {};

    interface InterfaceName {
        // Attributes
        attribute string name;

};
```

### 3. **ORB (Object Request Broker)**

#### **Definition**
The ORB is the middleware that handles communication between distributed objects. It manages all aspects of object communication.

#### **Key Responsibilities**
1. **Request Marshalling**: Convert parameters to network format (CDR)
2. **Request Transmission**: Send over IIOP protocol
3. **Request Unmarshalling**: Convert received data back to objects
4. **Exception Handling**: Propagate exceptions across network
5. **Service Location**: Find remote objects via Naming Service

#### **IIOP (Internet Inter-ORB Protocol)**
- **Protocol**: TCP/IP-based communication protocol
- **Default Port**: 1050 (configurable)
- **Message Format**: GIOP (General Inter-ORB Protocol)
- **Data Encoding**: CDR (Common Data Representation)

#### **CDR (Common Data Representation)**
- **Purpose**: Standardized binary encoding for data transmission
- **Alignment**: Handles byte alignment across platforms
- **Endianness**: Automatic big-endian/little-endian conversion
- **Encoding**: Compact binary format (more efficient than XML)

#### **ORB Initialization**
```java
ORB orb = ORB.init(args, null);
// args contains ORB configuration like:
// -ORBInitialPort 1050
// -ORBInitialHost localhost
```

---

### 4. **POA (Portable Object Adapter)**

#### **Definition**
POA is the component that manages the server-side object implementations (servants). It provides the bridge between CORBA objects and their implementations.

#### **Key Functions**
1. **Servant Management**: Creates and manages servant instances
2. **Activation**: Converts servants to CORBA object references
3. **Request Routing**: Directs incoming requests to appropriate servants
4. **Object Lifecycle**: Manages object creation and deletion

#### **Important Methods**
```java
// Create POA
POA rootpoa = POAHelper.narrow(orb.resolve_initial_references("RootPOA"));

// Activate POA Manager
rootpoa.the_POAManager().activate();

// Convert servant to CORBA object reference
org.omg.CORBA.Object ref = rootpoa.servant_to_reference(servant);

// Narrow to specific interface type
Calculator href = CalculatorHelper.narrow(ref);
```

#### **POA Advantages**
- **Portability**: Can use same servant code across ORB implementations
- **Flexibility**: Different activation and deactivation policies
- **Inheritance**: Servants inherit from POA skeletons

---

### 5. **Naming Service**

#### **Definition**
The CORBA Naming Service is a directory service that allows clients to locate remote objects by name.

#### **Hierarchy**
- **Root Context**: Top-level naming context
- **Naming Contexts**: Organize services hierarchically
- **Name Bindings**: Associate names with object references

#### **Operations**
```java
// Get Naming Service reference
org.omg.CORBA.Object objRef = orb.resolve_initial_references("NameService");
NamingContextExt ncRef = NamingContextExtHelper.narrow(objRef);

// Bind object (server-side)
NameComponent path[] = ncRef.to_name("Calculator");
ncRef.rebind(path, href);

// Resolve object (client-side)
Calculator calculator = CalculatorHelper.narrow(ncRef.resolve_str("Calculator"));
```

#### **Comparison with RMI Registry**
| Aspect | Naming Service | RMI Registry |
|--------|---|---|
| **Hierarchy** | Hierarchical | Flat |
| **Multiple services** | Yes | Single service per registry |
| **Standard** | CORBA standard | Java-specific |
| **Complexity** | Higher | Lower |

### 6. **Stubs and Skeletons**

#### **Stubs (Client-Side)**
- **Generated by IDL Compiler**: Creates proxy on client
- **Marshalling**: Serializes method parameters
- **Network Transmission**: Sends request to server
- **Unmarshalling**: Receives and deserializes response
- **Type Safety**: Provides Java interface to remote object

#### **Skeletons (Server-Side)**
- **Unmarshalling**: Deserializes incoming request
- **Method Dispatch**: Routes request to correct servant
- **Method Execution**: Calls actual servant implementation
- **Marshalling**: Serializes response
- **Transmission**: Sends response to client

### **File 1: Calculator.idl** (Interface Definition Language)

**Line-by-line Explanation**:

| Line | Code | Explanation |
|------|------|-------------|

- ✅ **Interface Contract**: Specifies exactly what methods available
- ✅ **Parameter Direction**: `in` indicates input parameters

- ✅ **Type System**: Strong typing (float, int, string, etc.)

### **File 4: CalculatorServer.java** (Server Process)

```java
import CalculatorApp.*;

import org.omg.CosNaming.*;

        }
    }

```

#### **Step 1: Initialize ORB**

**What happens**:

```
   -ORBInitialPort 1050       (port for IIOP)

2. Creates ORB instance
   
3. ORB initializes:
```

**Arguments Passed**:

#### **Step 2: Get Root POA and Activate**
```java

```

**What POA does**:
```
rootpoa = POAHelper.narrow(...)
    ↓
Resolves "RootPOA" from ORB
    ↓
Narrows CORBA.Object to POA interface
    ↓
POA now available for servant management

rootpoa.the_POAManager().activate()
    ↓
Activates POA Manager
    ↓
POA ready to receive and route requests
    ↓
Otherwise, incoming requests get no-response
```

#### **Step 3: Create Servant and Register with ORB**
```java

calculator.setORB(orb);
org.omg.CORBA.Object ref = rootpoa.servant_to_reference(calculator);
Calculator href = CalculatorHelper.narrow(ref);
```

**Execution Flow**:

```
1. new CalculatorImpl()
   └─ Creates servant instance (actual implementation)

2. calculator.setORB(orb)
   └─ Stores ORB reference for shutdown operations

3. rootpoa.servant_to_reference(calculator)
   └─ Converts servant object to CORBA object reference
   └─ POA assigns unique object ID
   └─ POA registers servant for request routing
   └─ Returns CORBA.Object reference

4. CalculatorHelper.narrow(ref)
   └─ Type-safe casting of CORBA.Object to Calculator
   └─ Ensures ref implements Calculator interface
   └─ Returns strongly-typed reference
   └─ Similar to Java instanceof + cast
```

**What POA Does Internally**:
```
Servant Instance: CalculatorImpl object
    ↓ POA creates mapping
{
    ObjectID: "xyz123"
    Servant: CalculatorImpl instance
    Interface: Calculator
}
    ↓
When request comes in with ObjectID "xyz123"
    ↓
POA routes to correct CalculatorImpl instance
    ↓
Request dispatched to mapped servant
```

#### **Step 4: Register with Naming Service**
```java
org.omg.CORBA.Object objRef = orb.resolve_initial_references("NameService");
NamingContextExt ncRef = NamingContextExtHelper.narrow(objRef);
NameComponent path[] = ncRef.to_name("Calculator");

```

**Execution Flow**:

```
1. orb.resolve_initial_references("NameService")
   └─ Asks ORB: "Where is the Naming Service?"
   └─ ORB has default location (usually localhost:1050)
   └─ Returns reference to Naming Service

2. NamingContextExtHelper.narrow(objRef)
   └─ Casts CORBA.Object to NamingContextExt
   └─ Provides directory service API

3. ncRef.to_name("Calculator")
   └─ Creates NameComponent path from string
   └─ Similar to file path "Calculator"
   └─ Could also be hierarchical: "Services/Calculator"

4. ncRef.rebind(path, href)
   └─ Registers mapping in Naming Service:
      Name: "Calculator"
      Reference: href (our remote object)
   └─ If already registered, overwrites (rebind vs bind)
```

**What Naming Service Now Contains**:
```
NameService Registry:
┌─────────────────────────────────┐
│ Name: "Calculator"              │
│ Reference: CORBA object ref     │
│ Interface: Calculator           │
│ Accessibility: Global           │
└─────────────────────────────────┘

```

#### **Step 5: Start Server Loop**
```java

```

**What `orb.run()` does**:
```
1. Starts main event loop
   
2. Loop continuously listens for:
   - Incoming IIOP requests on port 1050
   - Client method invocations
   - Naming Service queries

3. For each incoming request:
   - Unmarshal parameters (CDR → Java objects)
   - Determine target servant
   - Call appropriate method
   - Marshal result (Java object → CDR)
   - Send response

4. Loop continues indefinitely:
   - Server handles multiple clients
   - Each request in separate thread
   - Concurrent request processing

5. orb.run() blocks:
   - Main thread stays in loop
   - Server keeps running
   - Ctrl+C to terminate
```

```
    ↓
Server skeleton unmarshals and calls:
    CalculatorImpl.divide(10.0, 0.0)
    ↓
Server-side execution:
    if (b == 0)  // TRUE
        throw new DivisionByZero()
    ↓
Exception thrown on server
    ↓
Skeleton catches exception:
    Type: DivisionByZero
    ↓
Skeleton marshals exception:
    Exception type: DivisionByZero
    Exception fields: (none in this case)
    ↓
Skeleton creates exception response packet
    ↓
Send exception packet to client over IIOP
    ↓
Client stub receives exception packet
    ↓
Stub unmarshals exception:
    Exception type: DivisionByZero
    ↓
Stub throws exception:
    throw new DivisionByZero()
    ↓
Client code catches:
    catch (DivisionByZero e) {...}
    ↓
Display: "Error: Division by zero is not allowed!"
```

**Key Insight**: Exception thrown on server, serialized, transmitted over network, and re-thrown on client!

```
    - Error message displayed
```

```
│                  CORBA COMMUNICATION LAYERS              │
├──────────────────────────────────────────────────────────┤
│                                                          │
│  APPLICATION LAYER                                       │
│  (Your code: client.add(10, 5) or server.divide(x,y))   │

│         │ Marshalling              Unmarshalling        │
│         ↓                                ↓              │
│                                                          │
│  CORBA LAYER                                             │
│  Stubs (client)  ←→  POA/Skeletons (server)             │
│  Type narrowing, object references, method dispatch     │
│         ↑                                ↑              │
│         │ GIOP (General Inter-ORB Protocol)             │
│         ↓                                ↓              │
│                                                          │
│  TRANSPORT LAYER                                         │
│  IIOP (Internet Inter-ORB Protocol)                      │
│  TCP/IP sockets, network transmission                    │
│  Default port: 1050                                      │
│         ↑                                ↑              │
│         │ Binary (CDR format)                            │
│         ↓                                ↓              │
│                                                          │
│  NETWORK LAYER                                           │
│  TCP/IP protocol, routers, switches                      │
│                                                          │
└──────────────────────────────────────────────────────────┘
```

## 🎯 KEY DIFFERENCES: CORBA vs RMI

| Aspect | CORBA | RMI |
|--------|-------|-----|
| **Language Support** | C++, Java, Python, C#, ... | Java only |
| **Platform Independence** | OS independent | OS independent (via JVM) |
| **Interface Definition** | IDL (separate language) | Java interfaces |
| **Compilation** | IDL → compiler generates code | Direct Java compilation |
| **Service Discovery** | CORBA Naming Service | RMI Registry |
| **Default Port** | 1050 (IIOP) | 1099 |
| **Marshalling Format** | CDR (binary) | Java serialization |
| **Network Protocol** | IIOP (standardized) | Custom RMI protocol |
| **Learning Curve** | Steep | Easy |
| **Scalability** | Enterprise scale | Medium scale |
| **Industry Adoption** | Legacy systems | Declining |
| **Best For** | Heterogeneous enterprise | Java-to-Java |

---

## 🎯 KEY TAKEAWAYS

### **CORBA Advantages**:
1. ✅ Language-independent (Java, C++, Python, C#)
2. ✅ Platform-independent
3. ✅ Industry standard (OMG specification)
4. ✅ Mature technology (proven in enterprise)
5. ✅ Hierarchical naming service
6. ✅ Strong exception handling

### **CORBA Limitations**:
1. ❌ Steep learning curve
2. ❌ Complex setup (IDL, compilation, ORB)
3. ❌ Slower than RMI
4. ❌ Network overhead (GIOP/IIOP)
5. ❌ Declining industry adoption
6. ❌ Verbose code

### **When to Use CORBA**:
- Legacy system integration
- Multi-language distributed systems
- C++ and Java interoperability
- Enterprise-scale systems
- When language independence is critical

### **When NOT to Use CORBA**:
- New projects (prefer gRPC/REST)
- Microservices (use REST/gRPC)
- Web applications (use REST APIs)
- High-frequency systems (latency-sensitive)
- Simple Java-to-Java communication (use RMI)
